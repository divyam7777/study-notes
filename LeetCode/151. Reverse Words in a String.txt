class Solution(object):
    def reverseWords(self, s):
        z = s.split(" ")
        ans = []
        z.reverse()
        for i in z:
            if i != "":
                ans.append(i)
        f = " ".join(ans) 
        return(f)

# split will split sentence and store all words in a list
# then we can use the .reverse function to reverse that list
# after reversing we join all elements of the list with space as separator using ' '.join().
